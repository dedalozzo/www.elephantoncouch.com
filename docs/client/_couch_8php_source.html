89  // The JSON encoding is necessary because we might have complex keys. A complex key is no more than an array.
90  private function addMissingRows($keys, &$rows) {
91 
92  if (!empty($keys) && isset($rows)) {

95  $matches = [];
96  foreach ($rows as $row) {

197  public function validateAndEncodeDbName(&$name) {
198  # \A[a-z][a-z\d_$()+-/]++\z
199  #
200  # Assert position at the beginning of the string «\A»
201  # Match a single character in the range between “a” and “z” «[a-z]»
202  # Match a single character present in the list below «[a-z\d_$()+-/]++»
203  # Between one and unlimited times, as many times as possible, without giving back (possessive) «++»
204  # A character in the range between “a” and “z” «a-z»
205  # A single digit 0..9 «\d»
206  # One of the characters “_$()” «_$()»
207  # A character in the range between “+” and “/” «+-/»
208  # Assert position at the very end of the string «\z»
209  if (preg_match('%\A[a-z][a-z\d_$()+-/]++\z%', $name))
210  return $name = rawurlencode($name);
211  else
212  throw new \InvalidArgumentException("Invalid database name.");
